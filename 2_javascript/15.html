15. Sort objects by age property.
	var a = { name: "Вася", age: 23 };
	var b = { name: "Маша", age: 18 };
	var c = { name: "Вовочка", age: 6 };
	Answer:
		var result = [c,b,a];

function mySort() {
	var arr = [];
	for (var i = 0; i < arguments.length; i++) {
		arr.push(arguments[i]);
	}
	arr.sort(function(a, b) {
				return a.age - b.age;
			});
	return arr;
}
