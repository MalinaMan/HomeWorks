14. Sort array without native .sort() function.
	var arr = [5, 8, 31, 15125, 473, 12, 74, 1241, 0, 753, 0, -531, 151, 743, -35];

function mySort(arr) {
	var swapped;
	do {
		swapped = false;
		for (var i = 0; i < arr.length - 1; i++) {
			if (arr[i] > arr[i + 1]) {
				let temp = arr[i];
				arr[i] = arr[i + 1];
				arr[i + 1] = temp;
				swapped = true;
			}
		}
	} while (swapped);
	return arr;
}
