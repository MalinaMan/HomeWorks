10. Create function "find" that search value in array and return value's index or "-1";
	Example:
		var array = [1,23,4,5];
		find(23, array) //true
		find(3, array) //false

function find(needle, arr) {
	for (i = 0; i < arr.length; i++) {
		if (arr[i] === needle) {
			return i;
		}
	}
	return -1;
}
