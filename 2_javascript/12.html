12. Create function "filterRange(array,a,b)" that return new array with values between a,b params.
	Example:
		var arr = [5, 4, 3, 8, 0];
		var filtered = filterRange(arr, 3, 5); //filtered = [5, 4, 3]

function filterRange(array, a, b) {
	var arrNew = [];
	for (i = 0; i < array.length; i++) {
		if (array[i] >= a && array[i] <= b) {
			arrNew.push(array[i]);
		}
	}
	return arrNew;
}
